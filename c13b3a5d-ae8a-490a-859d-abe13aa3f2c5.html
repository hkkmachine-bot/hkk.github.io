         // 主倒數計時器
        function updateCountdown() {
            // 設置下一場演唱會的日期時間 (示例: 2023年12月15日 20:00:00)
            const concertDate = new Date("2025-06-23T11:00:00");
            const now = new Date();
            const diff = concertDate - now;

            if (diff <= 0) {
                document.getElementById('countdown').innerHTML = "搶票已開始！";
                return;
            }

            const days = Math.floor(diff / (1000 * 60 * 60 * 24));
            const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((diff % (1000 * 60)) / 1000);

            document.getElementById('countdown').innerHTML = 
                `${days}天 ${hours}小時 ${minutes}分鐘 ${seconds}秒`;
        }
        
        // 更新所有演唱會倒數

                document.getElementById(id).innerHTML = "搶票已開始！";
                return;
            }
            
            const days = Math.floor(diff / (1000 * 60 * 60 * 24));

            const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));

            document.getElementById(id).innerHTML = 
                `搶票倒數: ${days}天 ${hours}小時 ${minutes}分鐘 ${seconds}秒`;
        }
